import React, { useState, useEffect} from 'react';
import './App.css';
import axios from 'axios'; // Uvozimo Axios za slanje HTTP zahtjeva

function App() {
  const [toDos, setToDos] = useState([]);
  const [inProgress, setInProgress] = useState([]);
  const [done, setDone] = useState([]);
  const [showInput, setShowInput] = useState(false);
  const [newToDo, setNewToDo] = useState("");

  const handleButtonClick = () => {
    setShowInput(true);
  };

  const handleToDo = (event) => {
    setNewToDo(event.target.value);
  };

  const handleInputSubmit = () => {
    if(newToDo.trim() !== ""){
      setToDos([...toDos, newToDo]);
      setNewToDo("");
      setShowInput(false);
    }
  };

  const handleDragStart = (event, item) => {
    event.dataTransfer.setData("text/plain", item)
  }; 
  

  const handleDragOver = (event) => {
    event.preventDefault();
  };

  const handleDrop = (event, column) => {
    event.preventDefault();
    const item = event.dataTransfer.getData("text/plain");
    switch (column) {
      case "toDos":
        if (!toDos.includes(item)) {
          setToDos([...toDos, item]);
        }
        setInProgress(inProgress.filter(todo => todo !== item));
        setDone(done.filter(todo => todo !== item));
        break;
      case "inProgress":
        if (!inProgress.includes(item)) {
          setInProgress([...inProgress, item]);
        }
        setToDos(toDos.filter(todo => todo !== item));
        setDone(done.filter(todo => todo !== item));
        break;
      case "done":
        if (!done.includes(item)) {
          setDone([...done, item]);
        }
        setToDos(toDos.filter(todo => todo !== item));
        setInProgress(inProgress.filter(todo => todo !== item));
        break;
      default:
        break;
    }
  };

  return(
    <div className='app'>
      <div className='column'>
        <h2>To Do</h2>
        <div className='toDo'
        onDragOver={(event) => handleDragOver(event)}
        onDrop={(event) => handleDrop(event, "toDos")}>
          <button id="button-plus" onClick={handleButtonClick}>+</button>
          {showInput && (
            <div className='modal'>
              <div className='modal-content'>
                <textarea
                  type='text'
                  value={newToDo}
                  onChange={handleToDo}
                />
                <button onClick={handleInputSubmit}>Add</button>
              </div>
            </div>)}
          {toDos.map((item, index) => (
            <li
              key={index}
              draggable
              onDragStart={(event) => handleDragStart(event, item)}
              style={{marginTop: 30, marginLeft: 10}}
            >
              {item}
            </li>
          ))}
        </div>
      </div>

      <div className='column'>
        <h2>In Progress</h2>
        <div className='inProgress' 
        onDragOver={(event) => handleDragOver(event)}
        onDrop={(event) => handleDrop(event, "inProgress")}>
          {inProgress.map((item, index) => (
            <li
              key={index}
              draggable
              onDragStart={(event) => handleDragStart(event, item)}
              style={{marginTop: 30, marginLeft: 10}}
            >
              {item}
            </li>
          ))}
        </div>
      </div>

      <div className='column'>
        <h2>Done</h2>
        <div className='done'
        onDragOver={(event) => handleDragOver(event)}
        onDrop={(event) => handleDrop(event, "done")}>
          {done.map((item, index) => (
            <li
              key={index}
              draggable
              onDragStart={(event) => handleDragStart(event, item)}
              style={{marginTop: 30, marginLeft: 10, color: "green"}}
            >
              {item}
            </li>
          ))}
        </div>
      </div>
    </div>
  );
}

export default App;
